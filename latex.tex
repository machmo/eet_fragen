	\section{Das Drehstromsystem}
	\begin{itemize}
		\item Wie ist der Effektivwert einer periodisch veränderlichen Größe x(t) mit der Periodendauer T definiert? Welcher Zusammenhang besteht zwischen dem Effektivwert und dem Scheitelwert speziell bei sinusförmigen Größen?
		\ans{$U_{eff} = U = \sqrt{ \frac{1}{T} \int\limits_{t_0}^{t_0+T} u^2 (\tau) \, d \tau }$.\\Für sinusförmige Größen:
			$U_{eff} = \hat{u} \cdot \frac{1}{\sqrt{2}}$}
	
		\item Wie ist der Phasenverschiebungswinkel zwischen Strom und Spannung definiert?
		\ans{$\varphi = \varphi_{ui} = \varphi_u - \varphi_i $}
		
		\item Welche Funktion hat der Drehoperator $a$ und wie ist er im symmetrischen Drehstromsystem definiert?
		\ans{Hilfsmittel zur Darstellung der Phasenverschobenen Größen eines Drehstromsystems in Zeigerdarstellung.\\
			$\underline{a}^0 = 1,\; \underline{a} = e^{j\cdot 120^\circ},\; \underline{a}^2 = e^{j \cdot 240^\circ},\; \underline{a}^3 = 1$}
		
		\item In welchem Verhältnis stehen die Beträge der Strangströme eines Verbrauchers in Dreieckschaltung zu den Außenleiterströmen?
		\ans{$I_{L(\Delta)} = \sqrt{3} \cdot I_{\Delta}$}
		
		\item Vergleichen Sie Stern- und Dreiecksschaltung eines symmetrischen ohmschen Verbrauchers im Drehstromsytem:
		\begin{itemize}
		\item Wie ist das Verhältnis der Leistungsaufnahme bei gleichen Lastwiderständen?
		\item In welchem Verhältnis müssen die Lastwiderstände stehen, damit gleiche Leistung aufgenommen wird?
		\ans{Gleiche Lastwiderstände:\\
			$ P_{\Delta} = 3 \cdot P_S $\\
			Gleiche Leistung:\\
			$Z_{\Delta} = \frac{Z_S}{3} $}
		\end{itemize}
		
		\item Wie ist in einem einphasigen Wechselstromsystem die Schein-, Wirk- und Blindleistung definiert und mit welchen Einheiten werden sie bezeichnet?
		\ans{Wirkleistung\\
			$P= \Re{\{\underline{S}\}} = U\, I \cdot \cos(\varphi) \quad \text{[W]}$\\
			\\Blindleistung\\
			$Q=\Im{\{\underline{S}\}} = U \, I \cdot \sin(\varphi) \quad \text{[Var]}$\\
			\\Scheinleistung\\
			$S=U \cdot I = \sqrt{P^2 + Q^2} \quad \text{[VA]}$}
		
		\item Geben Sie die Formeln zur Berechnung der Schein-, Wirk- und Blindleistung in einem symmetrischen Drehstromsystem an.
		\ans{Wirkleistung\\
			$P= \Re{\{\underline{S}\}} = U\, I \cdot \cos(\varphi) \quad \text{[W]}$\\
			\\Blindleistung\\
			$Q=\Im{\{\underline{S}\}} = U \, I \cdot \sin(\varphi) \quad \text{[Var]}$\\
			\\Scheinleistung\\
			$S=U \cdot I = \sqrt{P^2 + Q^2} \quad \text{[VA]}$}
		
	\end{itemize}
